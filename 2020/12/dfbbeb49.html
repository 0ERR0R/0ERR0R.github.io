常用汇编指令

MOV指令

指令格式 指令解释
MOV r/m8,r8 r通用寄存器
MOV r/m16,r16 m代表内存
MOV r/m32,r32 imm代表立即数
MOV r8,r/m8 r8代表8位通用寄存器
MOV r16,r/m16 m8代表8位内存
MOV r32,r/m32 imm8代表8位立即数
MOV r8, imm8
MOV r16, imm16
MOV r32, imm32

ADD指令

指令格式

ADD r/m8, imm8
ADD r/m16,imm16
ADD r/m32,imm32
ADD r/m16,imm8
ADD r/m32,imm8
ADD r/m8,r8
ADD r/m16,r16
ADD r/m32,r32
ADD r8,r/m8
ADD r16,r/m16
ADD r32,r/m32

SUB指令(减法)

指令格式:

SUB r/m8, imm8
